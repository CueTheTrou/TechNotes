# Bash
pwd = path to working directory  
cd = change directory  
ls = list subfiles (prints all files in folder)  
.. = back a directory  
. = current directory  
/ = root of computer  
~ = user directory  
cat = to print the contents of a file to screen  
clear = clear terminal  
mkdir = make directory  
touch = creates file  
rm = remove  
rm -r = removes folder  
grep = filter output  
history = shows history of commands
